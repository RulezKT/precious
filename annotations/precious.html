This is a convenience script. Use eden(1) for greater convenience.

It takes json in various ways and calls precious.py with it. Run precious -h json or man precious-json or visit precious-json(7) for format details. The latter (web documentation) is better. Anything less (i.e. nothing) means the convenient defaults will be used.

man = (page, status = 0, cb) ->
  require('child_process').exec "man #{page}", (err, out) ->
    if err
      console.error "Can't get: `man #{page}`.".red
      console.error "Go to http://astrolet.github.com/precious/ for help."
      process.exit(1)
    else
      console.log "\n" + out
      cb() if cb?
      process.exit(status)

fetch = (what) ->
  stream = ephemeris convenient what
  stream.stderr.on 'data', (error) ->
    console.error '\nSpawned child_process error.\n'.red + error
    process.exit(1)
  stream.stdout.pipe process.stdout
